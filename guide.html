01개요

해당 탑승일의 예약 수요가 평소보다 높을 때, 차량선택 화면 하단에서 사용자의 예약 완료를 자연스럽게 유도하는 넛지 배너입니다. 진입 시 플로팅으로 강조했다가, 사용자가 화면과 상호작용하면 띠배너(도킹)로 가라앉아 방해를 최소화합니다.

트리거: 화면 진입 시노출: 진입할 때마다위치: CTA 바 바로 위

04인터랙션 · 모션

화면 진입 시(첫 진입·복귀 동일):

  • 바텀 바 뒤에서 슬라이드업 — 배너가 CTA 컨테이너 뒤(아래)에서 시작해 위로 올라와 도킹 위치에 안착 (CTA 바가 배너보다 위에 렌더 → 뒤에서 나오는 효과)
  • 도킹 → 플로팅 모프 — 도킹 스트립이 좌우 16 인셋·라운드 8·그라데이션·리프트되며 플로팅 카드로 전환
  • 플로팅 유지 — 쉬머 반복. 자동으로 도킹하지 않음
  • 도킹 — 사용자가 화면 터치·스크롤·클릭(첫 상호작용) 시 플로팅 → 도킹으로 가라앉음

타이밍 (1× 기준)

슬라이드업≈ 420ms · cubic-bezier(.4,0,.2,1)
도킹 유지≈ 160ms 후 모프
모프≈ 420ms (색·크기·위치·라운드 동시 전이)
도킹 복귀≈ 420ms (상호작용 트리거)
시작 오프셋translateY(74px) (바텀 바 뒤)

JS · 진입/도킹
function enter(){                         // 화면 진입마다 호출
  bar.classList.remove('is-floating');
  bar.style.transition = 'none';
  bar.style.opacity = '1';
  bar.style.transform = 'translateY(74px)';  // 바텀 바 뒤에서 시작
  bar.offsetWidth;                          // reflow
  bar.style.transition = '';
  bar.style.transform = 'translateY(0)';      // 슬라이드업 → 도킹
  setTimeout(() => {
    bar.classList.add('is-floating');       // 도킹 → 플로팅 모프
    armDock();
  }, 580);                                 // 슬라이드 + 유지
}
function armDock(){                       // 첫 상호작용에 도킹
  const dock = () => bar.classList.remove('is-floating');
  scroller.addEventListener('scroll', dock, { once:true, passive:true });
  screen.addEventListener('pointerdown', dock, { once:true });
}

enum NudgeState { case docked, floating }

struct NudgeInfoBar: View {
  let multiplier: Int                 // 실제 수요 배수
  @State private var state: NudgeState = .docked
  @State private var appeared = false

  var body: some View {
    text
      .padding(.horizontal, state == .floating ? 18 : 12)
      .padding(.vertical,   state == .floating ? 12 : 10)
      .frame(maxWidth: .infinity)
      .background(background)
      .clipShape(RoundedRectangle(cornerRadius: state == .floating ? 8 : 0))
      .shadow(color: Color(hex: "#32A2FA").opacity(state == .floating ? 0.28 : 0), radius: 12, y: 8)
      .padding(.horizontal, state == .floating ? 16 : 0)      // 사이드 마진
      .offset(y: appeared ? 0 : 74)                     // 바텀 바 뒤 → 슬라이드업
      .animation(.easeInOut(duration: 0.42), value: state)
      .onAppear { enter() }
  }

  private var text: some View {
    (Text("🔥 이 날은 예약이 평소 대비 ")
      + Text("\(multiplier)배").bold()
      + Text(" 몰리고 있어요"))
      .font(.system(size: 15)).lineLimit(1)
      .foregroundColor(state == .floating ? .white : Color(hex: "#383D4F"))
  }

  @ViewBuilder private var background: some View {
    if state == .floating {
      LinearGradient(colors: [Color(hex:"#0094F9"), Color(hex:"#32A2FA")],
                     startPoint: .leading, endPoint: .trailing)          // + ShimmerOverlay()
    } else { Color(hex: "#E2F2FE") }
  }

  private func enter() {
    withAnimation(.easeOut(duration: 0.42)) { appeared = true }    // 슬라이드업
    DispatchQueue.main.asyncAfter(deadline: .now() + 0.58) {
      state = .floating                                                    // 도킹 → 플로팅
    }
  }
}

enum class NudgeState { Docked, Floating }

@Composable
fun NudgeInfoBar(multiplier: Int, state: NudgeState, appeared: Boolean) {
  val floating = state == NudgeState.Floating
  val spec = tween<Float>(420, easing = FastOutSlowInEasing)

  val side  by animateDpAsState(if (floating) 16.dp else 0.dp, tween(420))
  val radius by animateDpAsState(if (floating) 8.dp else 0.dp, tween(420))
  val yOff  by animateDpAsState(if (appeared) 0.dp else 74.dp, tween(420))  // 슬라이드업
  val elev  by animateDpAsState(if (floating) 12.dp else 0.dp, tween(420))

  Row(
    Modifier
      .offset(y = yOff)
      .padding(horizontal = side)                          // 사이드 마진
      .fillMaxWidth()
      .shadow(elev, RoundedCornerShape(radius), spotColor = Color(0xFF32A2FA))
      .clip(RoundedCornerShape(radius))
      .background(if (floating) Brush.horizontalGradient(listOf(Color(0xFF0094F9), Color(0xFF32A2FA)))
                  else SolidColor(Color(0xFFE2F2FE))),
    horizontalArrangement = Arrangement.Center
  ) {
    Text(
      buildAnnotatedString {
        append("🔥 이 날은 예약이 평소 대비 ")
        withStyle(SpanStyle(fontWeight = FontWeight.Bold)) { append("${multiplier}배") }
        append(" 몰리고 있어요")
      },
      fontSize = if (floating) 15.sp else 13.sp, maxLines = 1,
      color = if (floating) Color.White else Color(0xFF383D4F)
    )
  }
}

@Composable
fun ShimmerOverlay(modifier: Modifier = Modifier) {
  val t = rememberInfiniteTransition()
  val p by t.animateFloat(0f, 1f, infiniteRepeatable(tween(1600, easing = LinearEasing)))
  Box(modifier.fillMaxSize().drawWithContent {
    drawContent()
    val w = 155.dp.toPx()
    val startX = -w + p * (size.width + 2 * w)
    drawRect(Brush.horizontalGradient(
      0f to Color.Transparent, 0.5f to Color.White.copy(alpha = .25f), 1f to Color.Transparent,
      startX = startX, endX = startX + w))
  })
}

문구: 🔥 이 날은 예약이 평소 대비 N배 몰리고 있어요

N (배수)해당 탑승일의 실제 수요 배수. 평소 기준(요일/시간대) 대비 서버 계산값.
‘이 날’이전 화면에서 선택한 탑승일 지칭 (오늘 아님).
강조‘N배’만 Bold (플로팅 흰색 / 도킹 blue-600).
⚠️ 데이터 무결성. N은 반드시 실데이터여야 합니다. 근거 없는 수치·상시 노출은 눈속임(다크패턴)으로 신뢰를 훼손하고 공정위 규제 리스크가 있습니다. 배수가 애매하거나 1.x배면 수치 없는 표현("예약이 빠르게 차고 있어요")으로 대체하세요.

## 목적
예약 3.0 "차량 선택" 화면 하단의 수요 넛지 배너. 해당 탑승일 예약 수요가 평소보다 높을 때
"🔥 이 날은 예약이 평소 대비 N배 몰리고 있어요"를 진입 시 플로팅으로 강조했다가,
사용자가 화면과 상호작용하면 도킹시켜 방해를 최소화한다. 외부 의존성 없는 단일 HTML 파일.

## 산출물 / 배포
- 소스: 단일 파일 index.html (HTML+CSS+JS 전부 인라인, 차량 이미지 assets/*.png만 외부)
- 로컬 프리뷰: 정적 서버로 서빙(예: node 정적 서버, 포트 4599) 후 브라우저에서 확인
- 배포: GitHub Pages(public repo). 라이브 = https://jonghyun-glitch.github.io/tada-rider-info-bar-prototype/
- 개발 가이드: guide.html (좌측에 index.html을 iframe로 실제 임베드, 우측 명세)

## 두 구현 버전 (⚙ 설정 토글로 전환)
1) 기존(legacy) — 넛지 단독. 진입: 바텀 바 뒤에서 슬라이드업 → 도킹 → 플로팅 모프.
   상호작용(스크롤/터치) 시 blue-50 풀블리드 띠배너로 도킹.
2) 신규(coexist) — ⚡"넥스트 호출 시 100% 예약 확정" 띠배너와 공존.
   진입: 바로 플로팅. 3초 후 첫 스크롤 → 플로팅이 '✓ 예약이 몰리는 날' 칩으로 모프(FLIP 축소·이동)
   → 이어서 ⚡ 100% 예약 확정 띠배너가 요금영역 위로 상승.
   칩 클릭 → 도킹 역순(undock)으로 플로팅 배너 복귀.

<script>
  const bar = document.getElementById('infobar');
  const tbanner = document.getElementById('tbanner');
  const nchip = document.getElementById('nchip');
  const tlEl = document.getElementById('tl');
  let impl='coexist', cond100=true, speed=0.75, disarmDock=null, morphTimer=null, armTimer=null;

  function applySpeed(){
    document.documentElement.style.setProperty('--dock-dur',(0.42/speed)+'s');
    document.documentElement.style.setProperty('--grad-dur',(1.6/speed)+'s');
  }
  const TL={
    legacy:'① 진입: 바텀 뒤 슬라이드업 → 도킹 → 플로팅<br>② 화면 터치·스크롤 시 blue-50 띠배너로 도킹',
    coexist:'① 진입: 플로팅 (⚡ 띠배너·칩 숨김)<br>② 3초 후 스크롤 → 플로팅이 ‘✓ 예약이 몰리는 날’ 칩으로 모프<br>③ 이어서 ⚡ 100% 예약 확정 띠배너 상승<br>④ 칩 클릭 → 역순으로 플로팅 배너 복귀'
  };
  function setTL(){ tlEl.innerHTML='<b>동작</b><br>'+TL[impl]; }
  function clearTimers(){ clearTimeout(morphTimer); clearTimeout(armTimer); if(disarmDock){disarmDock();disarmDock=null;} }

  function resetBase(){                                 // 바텀 바 뒤 시작 상태로 리셋
    clearTimers();
    bar.classList.remove('floating');
    bar.style.transition='none'; bar.style.opacity='1'; bar.style.visibility='visible';
    bar.style.transform='translateY(74px)'; bar.style.transformOrigin='';
    tbanner.classList.remove('show'); nchip.classList.remove('show');
    void bar.offsetWidth;
  }
  function armInteraction(cb, scrollOnly){              // 첫 상호작용에 1회 콜백
    if(disarmDock) disarmDock();
    const scroller=document.querySelector('.scroll'), phone=document.querySelector('.phone');
    const trigger=()=>{ disarm(); cb(); };
    function disarm(){ scroller.removeEventListener('scroll',trigger); scroller.removeEventListener('wheel',trigger); phone.removeEventListener('pointerdown',trigger); disarmDock=null; }
    scroller.addEventListener('scroll',trigger,{passive:true,once:true});
    scroller.addEventListener('wheel',trigger,{passive:true,once:true});
    if(!scrollOnly) phone.addEventListener('pointerdown',trigger,{once:true});
    disarmDock=disarm;
  }

  function dockLegacy(){ bar.classList.remove('floating'); }   // 기존: blue-50 스트립

  function dockCoexist(){                               // 신규: 플로팅 → ✓칩 모프 → ⚡띠배너 상승
    nchip.style.transition='none'; nchip.classList.add('show');   // 칩 최종 위치 사전 측정
    const to=nchip.getBoundingClientRect();
    nchip.classList.remove('show'); void nchip.offsetWidth; nchip.style.transition='';
    const from=bar.getBoundingClientRect();
    const dx=(to.left+to.width/2)-(from.left+from.width/2);
    const dy=(to.top+to.height/2)-(from.top+from.height/2);
    const sc=Math.max(to.width/from.width,.06);
    bar.style.transition='transform .42s cubic-bezier(.4,0,.2,1),opacity .32s ease';
    bar.style.transformOrigin='center center';
    setTimeout(()=>{                                    // rAF 대신 setTimeout (백그라운드 탭 rAF 정지 대비)
      bar.style.transform='translate('+dx+'px,'+dy+'px) scale('+sc+')';
      bar.style.opacity='0';
      nchip.classList.add('show');                     // 칩 등장
    }, 20);
    morphTimer=setTimeout(()=>{
      bar.style.visibility='hidden';
      if(cond100) tbanner.classList.add('show');        // ⚡ 띠배너 상승
    }, 470);
  }

  function undockCoexist(){                             // 칩 클릭 → 도킹 역순으로 플로팅 복귀
    if(impl!=='coexist' || !nchip.classList.contains('show')) return;
    clearTimers();
    tbanner.classList.remove('show');                   // ⚡ 띠배너 하강
    const from=nchip.getBoundingClientRect();           // 시작: 칩 위치
    bar.style.transition='none';                        // 끝: 플로팅 레이아웃 위치 측정
    bar.style.transform='translateY(0)'; bar.style.opacity='0';
    void bar.offsetWidth;
    const to=bar.getBoundingClientRect();
    const dx=(from.left+from.width/2)-(to.left+to.width/2);
    const dy=(from.top+from.height/2)-(to.top+to.height/2);
    const sc=Math.max(from.width/to.width,.06);
    bar.style.transformOrigin='center center';          // 칩 위치에 축소·투명 상태로 배치
    bar.style.transform='translate('+dx+'px,'+dy+'px) scale('+sc+')';
    bar.style.visibility='visible';
    void bar.offsetWidth;
    nchip.classList.remove('show');                     // 칩 사라짐
    bar.style.transition='transform .42s cubic-bezier(.4,0,.2,1),opacity .32s ease';
    setTimeout(()=>{                                     // 플로팅으로 확대·페이드인 (setTimeout: 백그라운드 rAF 대비)
      bar.style.transform='translateY(0)'; bar.style.opacity='1';
    }, 20);
    morphTimer=setTimeout(()=>{                          // 복귀 후 재도킹 재무장
      bar.style.transition=''; bar.style.transformOrigin='';
      armTimer=setTimeout(()=>armInteraction(dockCoexist,true), 3000);
    }, 470);
  }
  nchip.addEventListener('click', undockCoexist);

  function play(){
    resetBase(); setTL();
    bar.style.transition=''; bar.style.transform='translateY(0)';   // 슬라이드업
    if(impl==='legacy'){
      morphTimer=setTimeout(()=>{ bar.classList.add('floating'); armInteraction(dockLegacy,false); }, (420+160)/speed);
    } else {
      bar.classList.add('floating');                    // 신규: 바로 플로팅으로 상승
      armTimer=setTimeout(()=>armInteraction(dockCoexist,true), 3000);   // 3초 후 스크롤 도킹
    }
  }

  document.getElementById('impl').addEventListener('click',e=>{const b=e.target.closest('button');if(!b)return;
    impl=b.dataset.i; document.querySelectorAll('#impl button').forEach(x=>x.classList.toggle('on',x===b)); play();});
  document.getElementById('cond100').addEventListener('click',e=>{const b=e.target.closest('button');if(!b)return;
    cond100=b.dataset.c==='1'; document.querySelectorAll('#cond100 button').forEach(x=>x.classList.toggle('on',x===b)); play();});
  document.getElementById('replay').addEventListener('click',play);
  document.getElementById('replayFab').addEventListener('click',play);

  const panel=document.getElementById('panel'), scrim=document.getElementById('scrim');
  document.getElementById('fab').addEventListener('click',()=>{panel.classList.add('open');scrim.classList.add('open');});
  scrim.addEventListener('click',()=>{panel.classList.remove('open');scrim.classList.remove('open');});

  const phoneEl=document.getElementById('phone');

    document.querySelector('#screen-confirm .cf-scroll').scrollTop=0;
    phoneEl.classList.add('show-confirm');
  });
  document.getElementById('cf-back').addEventListener('click',()=>{
    resetBase();                          // 슬라이드 인 동안 바텀 바 뒤에 숨김
    phoneEl.classList.remove('show-confirm');
    setTimeout(play, 380);                // 슬라이드 인 후 재생
  });

  applySpeed();
  window.addEventListener('load',()=>setTimeout(play,400));
</script>
